public OnPlayerExitVehicle(playerid, vehicleid)
{
    if(IsASHCar(playerid))
	{
		if(GetPVarInt(playerid,"ASHGOTO") == 1)
		{
			SendClientMessage(playerid,0xFF6600AA, "Вы провалили практическую часть экзамена.");
			SetVehicleToRespawn(vehicleid);
			SetPVarInt(playerid,"ASHGOTO", 0);
			PlayerInfo[playerid][pASHG] = 0;
			PlayerInfo[playerid][pASHGO] = 0;
			DisablePlayerRaceCheckpoint(playerid);
			VehInfo[vehicleid][pEngine] = 0;
			VehInfo[vehicleid][pLights] = 0;
			VehInfo[vehicleid][pAlarm] = 0;
			VehInfo[vehicleid][pDoors] = 0;
			VehInfo[vehicleid][pBonnet] = 0;
			VehInfo[vehicleid][pBoot] = 0;
			VehInfo[vehicleid][pObjective] = 0;
			SetVehicleParamsEx(vehicleid,VehInfo[vehicleid][pEngine],VehInfo[vehicleid][pLights],VehInfo[vehicleid][pAlarm],0,VehInfo[vehicleid][pBonnet],VehInfo[vehicleid][pBoot],VehInfo[vehicleid][pObjective]);
		}
	}
	if(GetPlayerVehicleID(playerid) >= gruzcar[0] && GetPlayerVehicleID(playerid) <= gruzcar[4])
    {
        if(GetPVarInt(playerid, "loaderongoing") == 1)
        {
            DisablePlayerRaceCheckpoint(playerid);

        }
	}
	if(IsBusID(vehicleid))

		{
			PlayerInfo[playerid][CarJobID] = GetPlayerVehicleID(playerid);
			SetTimerEx("LeaveJob",20000,false,"i",playerid);
			SendClientMessage(playerid,0xFF6600AA, "У Вас есть 20 секунд чтобы вернуться в свой рабочий транспорт.");
		}
	}
	if(IsTaxiID(vehicleid))
	{
		if(PlayerInfo[playerid][TaxiTarif] > 0 && PlayerInfo[playerid][pJob] == 2)
		{
			PlayerInfo[playerid][CarJobID] = GetPlayerVehicleID(playerid);
			SetTimerEx("LeaveJob",20000,false,"i",playerid);

		}
	}
	if(IsAirID(vehicleid))
	{

public OnObjectMoved(objectid)
{
	new Float:x, Float:y, Float:z;
	for(new i; i < sizeof(Obj_FloorDoors); i ++)
	{
		if(objectid == Obj_FloorDoors[i][0])
		{
			GetObjectPos(Obj_FloorDoors[i][0], x, y, z);

			if(x < X_DOOR_L_OPENED - 0.5)   // Some floor doors have shut, move the elevator to next floor in queue:
			{
				Elevator_MoveToFloor(ElevatorQueue[0]);
				RemoveFirstQueueFloor();
			}
		}
	}
	if(objectid == Obj_Elevator)   // The elevator reached the specified floor.
	{
		KillTimer(ElevatorBoostTimer);  // Kills the timer, in case the elevator reached the floor before boost.

		FloorRequestedBy[ElevatorFloor] = INVALID_PLAYER_ID;

		Elevator_OpenDoors();
		Floor_OpenDoors(ElevatorFloor);

		GetObjectPos(Obj_Elevator, x, y, z);

		ElevatorState 	= ELEVATOR_STATE_WAITING;
		SetTimer("Elevator_TurnToIdle", ELEVATOR_WAIT_TIME, 0);
	}
	if(objectid==FerrisWheelObjects[10])SetTimer("RotateFerrisWheel",FERRIS_WHEEL_WAIT_TIME,false);
	return 1;
}
